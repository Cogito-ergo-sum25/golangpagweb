Table marcas {
  id_marca integer [primary key]
  nombre varchar(255)
  created_at timestamp
  updated_at timestamp
}

Table tipos_producto {
  id_tipo integer [primary key]
  nombre varchar(255)
  created_at timestamp
  updated_at timestamp
}

Table clasificaciones {
  id_clasificacion integer [primary key]
  nombre varchar(255)
  created_at timestamp
  updated_at timestamp
}

Table paises {
  id_pais integer [primary key]
  nombre varchar(100)
  codigo varchar(3)
}

Table certificaciones {
  id_certificacion integer [primary key]
  nombre varchar(255)
  organismo_emisor varchar(255)
  created_at timestamp
  updated_at timestamp
}

Table producto_certificaciones {
  id_producto integer [ref: > productos.id_producto]
  id_certificacion integer [ref: > certificaciones.id_certificacion]
  primary key (id_producto, id_certificacion)
}

Table productos {
  id_producto integer [primary key]
  id_marca integer [ref: > marcas.id_marca]
  id_tipo integer [ref: > tipos_producto.id_tipo]
  id_clasificacion integer [ref: > clasificaciones.id_clasificacion]
  id_pais_origen integer [ref: > paises.id_pais]
  
  sku varchar(255) [unique]
  nombre varchar(255) 
  nombre_corto varchar(255)
  modelo VARCHAR(100) 
  version varchar(255)
  serie VARCHAR(100)
  codigo_fabricante VARCHAR(100) 
  descripcion text
  
  imagen_url VARCHAR(255)
  ficha_tecnica_url VARCHAR(255)
  created_at timestamp
  updated_at timestamp
}
